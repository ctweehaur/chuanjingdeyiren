// 💡 终极修复版：加入对 • 符号的隔离机制，确保赏析长段落绝对对齐 function renderAnalysis() { const container = document.getElementById('analysis-content'); container.innerHTML = ''; analysisData.forEach(card => { // 1. 编译标题部分 const titlePyWords = card.py.split(/\s+/).filter(p => p.trim() !== ''); const titleZhChars = Array.from(card.header).filter(c => c.trim() !== ''); let titleRubyHtml = ''; titleZhChars.forEach((zh, idx) => { titleRubyHtml += `
${titlePyWords[idx] || ' '}
${zh}
`; }); // 2. 编译正文与列表行 let paragraphsHtml = ''; card.contents.forEach(line => { // 彻底过滤掉拼音序列中混入的标点和符号 const linePyWords = line.py.split(/\s+/).filter(p => p.trim() !== '' && !/[，。？！：“”‘’（）；、！,;.?!"'🏷️❶❷❸❹➔•·]/u.test(p)); const lineZhChars = Array.from(line.zh).filter(c => c.trim() !== ''); let lineRubyHtml = ''; let pyIdx = 0; lineZhChars.forEach((zh) => { // 💡 核心修复：判定是不是标点、数字、流向符号或列表圆点 • const isPuncOrSymbol = /[，。？！：“”‘’（）；、！,;.?!"'🏷️❶❷❸❹➔•·]/u.test(zh); let currentPy = ' '; if (!isPuncOrSymbol) { if (pyIdx < linePyWords.length) { currentPy = linePyWords[pyIdx]; pyIdx++; } } lineRubyHtml += `
${isPuncOrSymbol ? ' ' : currentPy}
${zh}
`; }); paragraphsHtml += `
${lineRubyHtml}
${line.en}
`; }); container.innerHTML += `
${titleRubyHtml}
${card.en}
${paragraphsHtml}
`; }); }
